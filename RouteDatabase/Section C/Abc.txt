#include <wiringPi.h>
#include <stdio.h>

#define PIN_SERVO 18

int main (void)
{
   printf ("Raspberry Pi wiringPi test program for servo SG90\n");
   
   wiringPiSetupGpio();
   
   pinMode (18, PWM_OUTPUT) ;
   
   pwmSetMode (PWM_MODE_MS);
   pwmSetRange (2000);
   pwmSetClock (192);
   pwmWrite(PIN_SERVO,150);
   
   delay(1000);
   
   pwmWrite(PIN_SERVO,200);
   

   

   
   return 0;
}
